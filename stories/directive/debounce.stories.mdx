export const Template3 = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  data() {
    return {
      time: new Date(),
      dur: '',
      inputStr: ''
    };
  },
  methods: {
    handleClick(){
      this.dur = (new Date() - this.time) / 1000 + 's'
      this.time = new Date()
    }
  },
  mounted(){
  },
  template:`
    <div>
      <input 
        v-debounce.keyup="[handleClick, funcDelay]"
        v-model="inputStr"
      />
      <br />
      时间间隔大于 {{funcDelay}} ms才能再次触发<br />
      按下间隔时间: {{dur}} 
    </div>
  `
});
